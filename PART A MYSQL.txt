CREATE TABLE customer (
    c_id      NUMERIC(6) NOT NULL COMMENT 'Customer Id',
    fname     VARCHAR(30) NOT NULL COMMENT 'First Name',
    lname     VARCHAR(30) NOT NULL COMMENT  'Last Name',
    st        VARCHAR(30) NOT NULL COMMENT 'Street Address',
    city      VARCHAR(30) NOT NULL COMMENT 'Name of the city',
    state     VARCHAR(2) NOT NULL COMMENT 'State Code',
    zipcode   NUMERIC(5) NOT NULL COMMENT 'Postal Code',
    gender    VARCHAR(1) COMMENT 'Gender',
    dob       DATE NOT NULL COMMENT 'Date of Birth',
    m_status  VARCHAR(1) NOT NULL COMMENT 'Marital Status',
    c_type    VARCHAR(2) NOT NULL COMMENT 'Customer Type (A / H / AH)'
);

ALTER TABLE customer ADD CONSTRAINT customers_pk PRIMARY KEY ( c_id );

CREATE TABLE drivers (
    license_no  VARCHAR(10) NOT NULL COMMENT 'Driver''s license number',
    d_fname     VARCHAR(30) NOT NULL COMMENT 'FIRST NAME',
    d_lname     VARCHAR(30) NOT NULL COMMENT 'LAST NAME',
    d_dob       DATE NOT NULL COMMENT 'Driver''s date of birth'
);

ALTER TABLE drivers ADD CONSTRAINT drivers_pk PRIMARY KEY ( license_no );

CREATE TABLE home (
    home_id        NUMERIC(8) NOT NULL COMMENT 'Unique Id for each home insured',
    purchase_date  DATE NOT NULL COMMENT 'Day the house was purchased',
    home_value     NUMERIC(10,2) NOT NULL COMMENT 'Home purchase value',
    area           NUMERIC(7, 2) NOT NULL COMMENT 'Area in Sqft',
    home_type      VARCHAR(1) NOT NULL COMMENT 'Type of Home',
    auto_fire      NUMERIC(1) NOT NULL COMMENT 'Auto Fire Notification',
    home_sec       NUMERIC(1) NOT NULL COMMENT 'Home Security System',
    pool           VARCHAR(1) COMMENT 'Swimming Pool',
    basement       NUMERIC(1) NOT NULL COMMENT 'DOES THE HOUSE HAVE A BASEMENT',
    policy_no      NUMERIC(12) COMMENT 'Policy Number'
);

ALTER TABLE home ADD CONSTRAINT home_pk PRIMARY KEY ( home_id );

CREATE TABLE invoice (
    invoice_id   NUMERIC(7) NOT NULL COMMENT 'INVOICE NUMBER FOR HOME POLICY',
    due_date     DATE NOT NULL COMMENT 'DAY PAYMENT IS DUE',
    invoice_amt  NUMERIC(10, 2) NOT NULL COMMENT 'AMOUNT DUE',
    policy_no    NUMERIC(12) NOT NULL COMMENT 'Policy Number'
);

ALTER TABLE invoice ADD CONSTRAINT invoice_pk PRIMARY KEY ( invoice_id );

CREATE TABLE payment (
    instal_id   NUMERIC(8) NOT NULL COMMENT 'INSTALLMENT ID',
    instal_amt  NUMERIC(10, 2) NOT NULL COMMENT 'AMOUT PAID IN INSTALMENT',
    pay_date    DATE NOT NULL COMMENT 'PAYMENT DATE',
    pay_method  VARCHAR(6) NOT NULL COMMENT 'Method of PAYMENT',
    invoice_id  NUMERIC(7) NOT NULL COMMENT 'INVOICE NUMBER'
);

ALTER TABLE payment ADD CONSTRAINT payment_pk PRIMARY KEY ( instal_id );

CREATE TABLE policy (
    policy_no   NUMERIC(12) NOT NULL COMMENT 'Policy Number',
    p_type      VARCHAR(1) NOT NULL COMMENT 'Primary Type (A/H)',
    c_id        NUMERIC(6) NOT NULL COMMENT 'Customer ID',
    start_date  DATE NOT NULL COMMENT 'POLICY START DATE',
    end_date    DATE NOT NULL COMMENT 'POLICY END DATE',
    premium     NUMERIC(10, 2) NOT NULL COMMENT 'MONTHLY PREMIUM AMOUNT',
    status      VARCHAR(1) NOT NULL COMMENT 'POLICY STATUS ( Current/Expired)'
);

ALTER TABLE policy ADD CONSTRAINT policy_pk PRIMARY KEY ( policy_no );

CREATE TABLE vehicle (
    vin        VARCHAR(10) NOT NULL COMMENT 'VEHICLE IDENTIFICATION NUMBER',
    v_make     VARCHAR(30) NOT NULL COMMENT 'VEHICLE MAKE AND MODEL',
    v_model    VARCHAR(30) NOT NULL COMMENT 'Vehicle Model',
    v_year     YEAR NOT NULL COMMENT 'YEAR THE CAR WAS BOUGHT',
    v_status   VARCHAR(1) NOT NULL COMMENT 'STATUS IS L,F,O',
    policy_no  NUMERIC(12) NOT NULL COMMENT 'POLICY NUMBER'
);

ALTER TABLE vehicle ADD CONSTRAINT vehicle_pk PRIMARY KEY ( vin );

CREATE TABLE vehicle_driver (
    vin         VARCHAR(10) NOT NULL COMMENT 'Vehicle Id Number',
    license_no  VARCHAR(10) NOT NULL COMMENT 'Driver''s license number',
    rating      NUMERIC(2) NOT NULL COMMENT 'DRIVER RATINGS'
);

ALTER TABLE vehicle_driver ADD CONSTRAINT vehicle_driver_pk 
    PRIMARY KEY ( vin, license_no );

ALTER TABLE home

ALTER TABLE payment

    ADD CONSTRAINT c_cust_name CHECK ( fname = UPPER(fname) AND lname = UPPER(lname)) ;

    ADD CONSTRAINT c_cust_city CHECK (city = UPPER(city));
